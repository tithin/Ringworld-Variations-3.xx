##############################
# traversal trance - intended to be used to make it easier to get to gateway activation
###############################
 traversal_trance = {
	resources = {
		category = artifact_actions
			cost = {
				minor_artifacts = 1500
			}
	}
	ai_weight = {
		weight = 100
	}

	potential = {
		has_technology = "tech_traversal_trance"
		NOT = { has_technology = tech_gateway_activation } ## { has_country_flag = traversal_trance_triggered }
	}

	allow = {
		custom_tooltip = {
			fail_text = "Your Scientists have already discovered this."
			NOT = { has_technology = tech_gateway_activation }
		}
	}

	effect = {
		custom_tooltip = traversal_trance_desc
		hidden_effect = {
			add_modifier = {
				modifier = "traversal_trance_decision_used" 
				days = 30
			}
		owner = {
			country_event = { id = traversal_trance.1 }
		}
 	}
}
}
